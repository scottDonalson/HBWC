package rocks.hbwc;


import rocks.hbwc.entities.Wrestler;
import rocks.hbwc.repository.WrestlerRepositoryImpl;

import javax.persistence.EntityManager;
import javax.persistence.EntityManagerFactory;
import javax.persistence.Persistence;
import java.util.Optional;

public class Application {

	public static void main(String[] args) {
		EntityManagerFactory entityManagerFactory = Persistence.createEntityManagerFactory("default");
		EntityManager entityManager = entityManagerFactory.createEntityManager();
		WrestlerRepositoryImpl wrestlerRepository = new WrestlerRepositoryImpl(entityManager);

		Wrestler wrestler = new Wrestler();
		wrestler.setFirstName("Zachary");
		wrestler.setLastName("Donalson");

		Optional<Wrestler> savedWrestler = wrestlerRepository.save(wrestler);

		entityManager.close();



	}

}
